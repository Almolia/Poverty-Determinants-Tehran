Project Setup: Libraries and Helper Functions

This initial cell sets up the analytical environment for the project. It performs two key tasks:

  1. Importing Libraries:
  • We import essential Python libraries for data manipulation (pandas, numpy), statistical modeling (statsmodels), and visualization (matplotlib, pygam).
  • Specific modules are imported to handle potential warnings and perform model diagnostics.
  1. Defining Helper Functions:

  • weighted_median: A function to accurately calculate the weighted median of household expenditure, necessary for defining the relative poverty line each year.
  • prepare_data: A comprehensive function to load, clean, and preprocess the raw data. This includes calculating the target variable (Is_Poor), standardizing numerical features, and setting reference levels for categorical variables.
  • fit_and_report_model: A function to fit the logistic regression model and generate a detailed report, including coefficients, odds ratios, marginal effects, VIF scores, and goodness-of-fit statistics.
  • run_analysis: The main wrapper function that orchestrates the entire analysis pipeline.

def hosmer_lemeshow_test(results, df, g=10):
    try:
        endog_name = results.model.endog_names
        df['prob'] = results.predict()
        df['decile'] = pd.qcut(df['prob'], g, labels=False, duplicates='drop')
        
        # Use survey weights in H-L calculation
        df['analysis_weights'] = df.get('analysis_weights', 1.0)
        
        # Calculate observed and expected frequencies in a vectorized way
        df['observed_poor_w'] = df[endog_name] * df['analysis_weights']
        df['expected_poor_w'] = df['prob'] * df['analysis_weights']
        df['observed_not_poor_w'] = (1 - df[endog_name]) * df['analysis_weights']
        df['expected_not_poor_w'] = (1 - df['prob']) * df['analysis_weights']

        # Group by decile and sum to get the observed and expected counts per group
        summary = df.groupby('decile')[[
            'observed_poor_w', 'expected_poor_w',
            'observed_not_poor_w', 'expected_not_poor_w'
        ]].sum()

        # Calculate the Hosmer-Lemeshow chi-squared statistic
        hl_stat = (((summary['observed_poor_w'] - summary['expected_poor_w'])**2 / summary['expected_poor_w']) + 
                   ((summary['observed_not_poor_w'] - summary['expected_not_poor_w'])**2 / summary['expected_not_poor_w'])).sum()
        
        df_hl = g - 2
        p_value = 1 - chi2.cdf(hl_stat, df_hl)
        return pd.Series({'Chi-squared': hl_stat, 'p-value': p_value, 'df': df_hl})
    
    except Exception as e:
        # Provide a more informative error message if the test fails
        return f"Could not perform H-L test. Reason: {e}"
    
def weighted_median(data, weights):
    df_temp = pd.DataFrame({'data': data, 'weights': weights}).dropna()
    df_temp = df_temp.sort_values('data')
    df_temp['cumulative_weight'] = df_temp['weights'].cumsum()
    total_weight = df_temp['weights'].sum()
    median_value = df_temp['data'][df_temp['cumulative_weight'] >= total_weight / 2].iloc[0]
    return median_value

def prepare_data(df, pp):
    # This function remains the same as your original
    df.loc[:, 'Equivalized_Net_Expenditure'] = df['Net_Expenditure'] / df['OECD_Modified']
    df = df.drop(columns=['Net_Expenditure', 'OECD_Modified'])
    df = df.dropna()
        
    yearly_dfs = []
    for year in sorted(df['Year'].unique()):
        df_year = df[df['Year'] == year].copy()
        median_expenditure = weighted_median(
            data=df_year['Equivalized_Net_Expenditure'],
            weights=df_year['Household_Weight']
        )
        poverty_line = pp * median_expenditure
        print(f"Poverty line for year {year} ({int(pp*100)}%): {poverty_line:,.2f}")
        df_year['Is_Poor'] = (df_year['Equivalized_Net_Expenditure'] < poverty_line).astype(int)
        yearly_dfs.append(df_year)
    df = pd.concat(yearly_dfs)

    numerical_features = ['Aid_Income_Share', 'Labor_Income_Share', 'Pension_Income_Share', 'Capital_Income_Share', 'Imputed_Rent_Share', 'Other_NonCash_Income_Share', 'Dependency_Ratio_Age', 'Dependency_Ratio_Employment', 'Head_Age_']
    scaler = StandardScaler()
    df[numerical_features] = scaler.fit_transform(df[numerical_features])

    def set_reference_level(df, column, ref_level):
        if df[column].dtype.name != 'category':
            df[column] = pd.Categorical(df[column])
        current_cats = df[column].cat.categories.tolist()
        if ref_level in current_cats:
            current_cats.remove(ref_level)
            new_order = [ref_level] + current_cats
            df[column] = df[column].cat.reorder_categories(new_order, ordered=False)
        return df

    df = set_reference_level(df, 'Head_Education_Level', 'Diploma')
    df = set_reference_level(df, 'Head_Activity_Status', 'Employed')
    df = set_reference_level(df, 'Head_Marital_Status', 'Married')
    df = set_reference_level(df, 'Head_Sex', 'Male')
    df = set_reference_level(df, 'Head_Age', '40-50')
    df = set_reference_level(df, 'Tenure_Type', 'Owner')
    df = set_reference_level(df, 'Urban_Rural', 'Rural')

    return df

def fit_and_report_model(df, formula, model_name, alpha):
    print(f"{'='*120}")
    
    try:
        df['analysis_weights'] = df['Household_Weight'] * (len(df) / df['Household_Weight'].sum())
        
        model = smf.logit(formula=formula, data=df)
        results = model.fit(cov_type='HC1', wls_weights=df['analysis_weights'])
        
        # Get the design matrix from the fitted model
        R = results.model.exog

        # Create a DataFrame for VIF calculation
        vif_data = pd.DataFrame()
        vif_data["feature"] = results.model.exog_names
        vif_data["VIF"] = [variance_inflation_factor(R, i) for i in range(R.shape[1])]

        print(f"{'='*120}\n")
        print(vif_data)
        
        # --- Create a comprehensive results DataFrame ---
        params = results.params
        conf = results.conf_int()
        p_values = results.pvalues
        
        # Get marginal effects as a dataframe
        margeff_df = results.get_margeff().summary_frame()

        # Combine results into one DataFrame
        results_df = pd.DataFrame({
            "Coefficient": params,
            "p_value": p_values,
            "Conf. Int. [0.025]": conf[0],
            "Conf. Int. [0.975]": conf[1],
            "Odds_Ratio": np.exp(params),
            "OR_CI_2.5%": np.exp(conf[0]),
            "OR_CI_97.5%": np.exp(conf[1]),
        })
        
        # Join marginal effects
        final_results_df = results_df.join(margeff_df[['dy/dx']])
        final_results_df.rename(columns={'dy/dx': 'Marginal_Effect'}, inplace=True)
        
        # --- Filter by significance ---
        alpha = alpha  # Significance level
        significant_df = final_results_df[final_results_df['p_value'] < alpha]
        non_significant_df = final_results_df[final_results_df['p_value'] >= alpha]

        # --- Report Non-Significant Features ---
        print(f"\n{'='*120}\n--- Non-Significant Features (p >= {alpha}) ---")
        non_sig_vars = non_significant_df.drop('Intercept', errors='ignore')
        if non_sig_vars.empty:
            print("All features are statistically significant.")
        else:
            print(non_sig_vars[['p_value']].to_string(float_format="%.3f"))
        print(f"{'='*120}")
        
        # --- Report Significant Features in Sorted Tables ---
        print(f"\n--- Significant Features Report (p < {alpha}) ---\n")

        # 1. Sorted
        sorted_by_coeff = significant_df.sort_values('Coefficient', ascending=False).drop('Intercept', errors='ignore')
        print(sorted_by_coeff[['Coefficient', 'Odds_Ratio', 'Marginal_Effect', 'p_value']].to_string(float_format="%.4f"))
        print(f"\n{'='*120}")

        # --- 3. Goodness-of-Fit Report (Now it will work) ---
        print("--- Goodness-of-Fit Statistics ---")
        print(f"Observations:          {int(results.nobs):,}")
        print(f"Pseudo R-squared:      {results.prsquared:.4f} (McFadden's)")
        print(f"Log-Likelihood:        {results.llf:,.2f}")
        print(f"AIC:                   {results.aic:,.2f}")
        print(f"BIC:                   {results.bic:,.2f}")
        
        # Hosmer-Lemeshow Test
        hl_results = hosmer_lemeshow_test(results, df.copy())
        print("Hosmer-Lemeshow Test:")
        print(hl_results.to_string(float_format="%.10f"))
        print(f"{'='*120}")
        
    except Exception as e:
        print(f"Could not fit model '{model_name}'. Error: {e}")

def run_analysis(df, poverty_thresholds, alpha, controls, demographics, household_structure):    

    models_to_run = {
        "First Model": 
            f"Is_Poor ~ {demographics} + {household_structure} + {controls}"}

    for pp in poverty_thresholds:
        print(f"{'='*120}\nANALYSIS FOR POVERTY LINE: {int(pp*100)}% of Median by Alpha of {alpha}\n{'='*120}")
        df_prepared = prepare_data(df, pp).drop_duplicates(subset=['ID'])
        for name, formula in models_to_run.items():
            fit_and_report_model(df_prepared, formula, name, alpha)

Step 5: Semi-Parametric Modeling - Generalized Additive Model (GAM)

The final logistic regression model failed the Hosmer-Lemeshow goodness-of-fit test, indicating that the model's linear assumptions do not fully capture the underlying data patterns. To overcome this limitation, we now fit a Generalized Additive Model (GAM).

GAMs extend generalized linear models by allowing for non-linear relationships between predictors and the outcome. This is achieved using:

  • Smooth terms s(): For continuous variables, allowing the data to determine the shape of the relationship.
  • Factor terms f(): For categorical variables.

We use a regularized model and employ grid search (gridsearch) to find the optimal smoothing parameter (lambda), preventing overfitting. The following code fits the GAM and generates Partial Dependence Plots, which visualize the marginal effect of each variable on the log-odds of poverty while holding other variables constant. These plots will reveal the complex, non-linear dynamics that the logistic model could not.

def weighted_median(data, weights):
    df_temp = pd.DataFrame({'data': data, 'weights': weights}).dropna()
    df_temp = df_temp.sort_values('data')
    df_temp['cumulative_weight'] = df_temp['weights'].cumsum()
    total_weight = df_temp['weights'].sum()
    median_value = df_temp['data'][df_temp['cumulative_weight'] >= total_weight / 2].iloc[0]
    return median_value

def prepare_data(pp=0.6):
    df = pd.read_csv("Tehran_all_prepared.csv")
    
    # Ensure correct data types for calculation
    df['Net_Expenditure'] = pd.to_numeric(df['Net_Expenditure'], errors='coerce')
    df['OECD_Modified'] = pd.to_numeric(df['OECD_Modified'], errors='coerce')

    df.loc[:, 'Equivalized_Net_Expenditure'] = df['Net_Expenditure'] / df['OECD_Modified']
    df = df.drop(columns=['Net_Expenditure', 'OECD_Modified'])
    df = df.dropna(subset=['Equivalized_Net_Expenditure', 'Household_Weight'])

    # Yearly loop to define poverty
    yearly_dfs = []
    for year in sorted(df['Year'].unique()):
        df_year = df[df['Year'] == year].copy()
        median_expenditure = weighted_median(data=df_year['Equivalized_Net_Expenditure'], weights=df_year['Household_Weight'])
        poverty_line = pp * median_expenditure
        df_year['Is_Poor'] = (df_year['Equivalized_Net_Expenditure'] < poverty_line).astype(int)
        yearly_dfs.append(df_year)
    df = pd.concat(yearly_dfs)

    # Define features
    numerical_features = ['Aid_Income_Share', 'Labor_Income_Share', 'Pension_Income_Share', 'Capital_Income_Share', 'Imputed_Rent_Share', 'Other_NonCash_Income_Share', 'Dependency_Ratio_Employment', 'Head_Age_']
    categorical_features = ['Head_Education_Level', 'Head_Activity_Status', 'Tenure_Type', 'Urban_Rural']
    
    # Drop rows with NaNs in feature columns
    df = df.dropna(subset=numerical_features + categorical_features)

    # Scale numerical features
    scaler = StandardScaler()
    df[numerical_features] = scaler.fit_transform(df[numerical_features])
    
    # Set categorical features
    for col in categorical_features:
        df[col] = pd.Categorical(df[col])

    return df, scaler, numerical_features, categorical_features

def plot_gam_results(gam, df, features, numerical_features, scaler):
    """
    Generates and saves partial dependence plots for a fitted GAM model.
    - Correctly handles numerical vs. categorical plotting logic.
    - Simplifies and corrects the logic for categorical features.
    - Sorts categorical features by their effect for better interpretability.
    """
    print("\nGenerating partial dependence plots with LOG-ODDS axes...")

    n_features = len(features)
    n_cols = 3
    n_rows = int(np.ceil(n_features / n_cols))

    fig, axs = plt.subplots(n_rows, n_cols, figsize=(n_cols * 6, n_rows * 5))
    fig.suptitle('Regularized GAM Partial Dependence Plots', fontsize=18, y=1.02)

    axs = axs.flatten()

    for i, feature_name in enumerate(features):
        ax = axs[i]

        if feature_name in numerical_features:
            # --- Logic for Numerical Features (Splines) ---
            XX = gam.generate_X_grid(term=i, n=100)
            pdep, confi = gam.partial_dependence(term=i, X=XX, width=0.95)
            pdep = np.exp(pdep)
            confi = np.exp(confi)

            # Create a dummy array to inverse-transform the feature's values
            # to their original scale for plotting.
            dummy_scaled_data = np.zeros((XX.shape[0], len(numerical_features)))
            feature_idx_in_scaler = numerical_features.index(feature_name)
            dummy_scaled_data[:, feature_idx_in_scaler] = XX[:, i]
            unscaled_full_data = scaler.inverse_transform(dummy_scaled_data)
            unscaled_x = unscaled_full_data[:, feature_idx_in_scaler]

            ax.plot(unscaled_x, pdep, color='darkblue', linewidth=2.5)
            ax.fill_between(unscaled_x.ravel(), confi[:, 0], confi[:, 1], color='lightblue', alpha=0.4)
            ax.set_xlabel(feature_name.replace('_', ' '), fontsize=12) # Set x-label for numerical

        else:
            # --- CORRECTED Logic for Categorical Features (Factors) ---
            # 1. Get the original category names. They are ordered in a way that
            #    corresponds to the integer codes (0, 1, 2, ...) used by the model.
            categories = df[feature_name].cat.categories

            # 2. Generate a grid for the feature and compute its partial dependence.
            #    The results (pdep, confi) are ordered by the category codes.
            XX = gam.generate_X_grid(term=i)
            pdep, confi = gam.partial_dependence(term=i, X=XX, width=0.95)

            # 3. Create a DataFrame to hold the results, linking category names to their effects.
            effect_df = pd.DataFrame({
                'category': categories,
                'pdep': pd.unique(pdep),
                'confi_lower': pd.unique(confi[:, 0]),
                'confi_upper': pd.unique(confi[:, 1])
            })
            cols_to_transform = ['pdep', 'confi_lower', 'confi_upper']
            effect_df[cols_to_transform] = effect_df[cols_to_transform].apply(np.exp)

            if feature_name in ['Head_Education_Level', 'Head_Activity_Status', 'Tenure_Type']:
                effect_df = effect_df.sort_values('pdep', ascending=False) # Sort by effect size for interpretability

            # 4. Plot using a simple integer range for the x-axis.
            x_range = np.arange(len(effect_df))
            ax.plot(x_range, effect_df['pdep'], 'o-', color='darkblue', linewidth=2.5, markersize=8)
            ax.fill_between(x_range, effect_df['confi_lower'], effect_df['confi_upper'], color='lightblue', alpha=0.4)

            # 5. Set the ticks and labels using the sorted category names.
            ax.set_xticks(x_range)
            ax.set_xticklabels(effect_df['category'], rotation=45, ha='right', fontsize=10)
            ax.set_xlabel('') # Clear x-label for categorical for tidiness

        ax.set_title(feature_name.replace('_', ' '), fontsize=14)
        ax.set_ylabel('Odds Ratio on Poverty', fontsize=12)
        ax.axhline(y=1, linestyle='--', color='red', linewidth=1)
        ax.grid(True, linestyle='--', alpha=0.6)

    # Hide any unused subplots
    for j in range(i + 1, len(axs)):
        axs[j].set_visible(False)

    plt.tight_layout(rect=[0, 0, 1, 0.98])
    plt.show()

# Main Analysis Function (Corrected and Integrated)
# --------------------------------------------------------------------
def generate_terms(features, numericals):
    gam_terms = []
    for i, feature in enumerate(features):
        if feature in numericals:
            gam_terms.append(s(i))
        else:
            gam_terms.append(f(i))
    
    # Start with the first term and add the rest
    gam_formula = gam_terms[0]
    for i in range(1, len(gam_terms)):
        gam_formula += gam_terms[i]
    
    return gam_formula

def run_gam_analysis():
    print("="*80)
    print("Running REGULARIZED GAM Analysis")
    print("="*80)
    
    df, scaler, numerical_features, categorical_features = prepare_data()
    
    features = numerical_features + categorical_features
    
    X = df[features].copy()
    # Convert only categorical columns to codes for the model
    for col in categorical_features:
        X[col] = X[col].cat.codes

    y = df['Is_Poor'].values
    weights = df['Household_Weight'].values

    # --- Build the formula dynamically ---
    gam_formula = generate_terms(features, numerical_features)

    print("\nFitting GAM model using gridsearch for optimal smoothing...")
    lams = np.exp(np.random.random((15, len(features))) * 6 - 3)
    gam = LogisticGAM(gam_formula).gridsearch(X.values, y, weights=weights, lam=lams)
    print("Model fitting complete.\n")
    
    gam.summary()
    
    return gam, df, features, numerical_features, scaler
